

import csv  # CSV 파일을 읽기 위해 'csv' 모듈을 가져옵니다.

#- 'age.csv' 파일을 cp949 인코딩(한글 처리에 주로 사용)으로 읽기 모드('r'이 기본값)로 엽니다.
f = open('age.csv', encoding='cp949')
#- 열린 파일 객체 'f'를 csv.reader 객체로 변환하여 'data'에 저장합니다.
#- 이제 'data'는 반복 가능한 CSV 데이터 객체가 됩니다.
data = csv.reader(f)

#- 'next(data)'를 사용하여 'data'의 첫 번째 줄(헤더/컬럼 이름)을 건너뛰고 'header' 변수에 저장합니다.
#- 이후 'data' 반복은 두 번째 줄(실제 데이터)부터 시작합니다.
header = next(data)

#- 최종 결과를 저장할 빈 리스트 'result'를 초기화합니다.
result = []

#- 'data' 객체(헤더를 제외한 실제 데이터 행들)를 순회합니다.
for row in data:
    # 현재 행(row)의 첫 번째 컬럼(row[0])에 '신도림'이라는 문자열이 포함되어 있는지 확인합니다.
    if '신도림' in row[0]:
        # '신도림'이 포함된 경우, 해당 행의 네 번째 컬럼(인덱스 3)부터 끝까지 순회합니다.
        # 이 부분은 주로 연령별 인구수 데이터가 있는 컬럼들일 가능성이 높습니다.
        for i in row[3:]:
            # 순회하는 각 데이터 'i'를 정수(int)로 변환한 후 'result' 리스트에 추가합니다.
            result.append(int(i))

#- 파일 사용이 끝났으므로 파일 객체를 닫아줍니다. (중요)
f.close()

#- 최종적으로 '신도림' 지역의 인구수 데이터(정수형)가 저장된 'result' 리스트를 출력합니다.
print(result)
